455-分发饼干

  • 很简单的贪心，用尽可能小的饼干来满足每个学生（局部贪心），充分利用饼干资源，就能满足尽可能多的学生（全局最优）。
class Solution {
public:
    int findContentChildren(vector<int>& g, vector<int>& s) {
        sort(g.begin(), g.end());
        sort(s.begin(), s.end());
        int cnt = 0;
        auto jt = s.begin();
        for (auto it = g.begin(); it < g.end(); it++)
        {
            while (jt != s.end() && *jt < *it)
            {
                jt ++;
            }
            if(jt == s.end()) break;
            jt ++;
            cnt ++;
        }
        return cnt;
    }
};

860-柠檬水找零

  • 分类讨论即可。
  • 5元的找零能力强，就尽可能每次找零留下5用10找（局部贪心），就能保证全局性质上的最佳找零能力。如果全局性质上的最佳找零能力都满足不了需求，说明确实无法完成该bills数列的找零。
class Solution {
public:
    bool lemonadeChange(vector<int>& bills) {
        unordered_map<int, int> m;
        for (auto it = bills.begin(); it < bills.end(); it++)
        {
            if(*it == 5) {
                m[5] ++;
            }else if (*it == 10)
            {
                if(m[5] > 0) {
                    m[5] --;
                    m[10] ++;
                }
                else {
                    return false;
                }
            }else {
                if(m[5] > 0 && m[10] > 0) {
                    m[5] --;
                    m[10] --;
                }
                else if (m[5] >= 3)
                {
                    m[5] -= 3;
                }
                else {
                    return false;
                }
            }
        }
        return true;
    }
};

56-合并区间

  • 前探式，不断吸收有重叠的区间merge成一个。
  • 主要是需要先排序，需要自己重载一下cmp函数，而且是static函数。
class Solution {
public:
    static bool cmp(vector<int> a, vector<int> b) {
        return a[0] < b[0];
    }

    vector<vector<int>> merge(vector<vector<int>>& intervals) {
        if(intervals.size() == 1) return intervals;
        sort(intervals.begin(), intervals.end(), cmp);
        // 循环入口设计想清楚再写
        auto it = intervals.begin();
        auto itt = it + 1;
        while (itt != intervals.end())
        {
            if((*it)[1] >= (*itt)[0]) {
                if((*it)[1] <= (*itt)[1]) (*it)[1] = (*itt)[1]; // 后尾长于前尾
                intervals.erase(itt);
            }else {
                it ++;
            }
            itt = it + 1;
        }
        return intervals;
    }
};

57-插入区间

  • 核心思路就是单独分一个res来保存结果，直接在intervals里处理复杂度过高。
  • 插入区间之前的区间直接放入res，被插入区间影响到的区间和插入区间一起打包merge，插入区间后的区间也直接放入res。
  • 比较麻烦的是三种特殊情况：直接塞到最前面，直接塞到最后面和插在缝里，这几种都没有产生影响区间。
class Solution {
public:
    vector<vector<int>> insert(vector<vector<int>>& intervals, vector<int>& newInterval) {
        if(intervals.empty()) {
            intervals.push_back(newInterval);
            return intervals;
        }
        vector<vector<int>> res;
        auto it = intervals.begin();
        if((*it)[0] > newInterval[1]) { // 直接塞到最前
            intervals.insert(it, newInterval);
            return intervals;
        }
        while ((*it)[1] < newInterval[0])
        {
            res.push_back((*it));
            it ++;
            if(it == intervals.end()) {
                intervals.push_back(newInterval);
                return intervals;
            }; // 没有一个被影响的，直接塞到尾部
        }
        // (*it)[1] >= newInterval[0] 当下it已经被影响？不一定
        if((*it)[0] > newInterval[1]) {
            intervals.insert(it, newInterval); // 插入缝里
            return intervals;
        }
        auto start = it;// 影响的起始区间
        auto end = it;
        while ((*it)[0] <= newInterval[1])
        {
            end = it;
            it ++;
            if(it == intervals.end()) { //影响到尾部
                break;
            }
        }
        // 当下it已不被影响 end是终止区间
        vector<int> merge = {min(newInterval[0], (*start)[0]), max(newInterval[1], (*end)[1])};
        res.push_back(merge);
        while (it != intervals.end()) //加入剩余的区间
        {
            res.push_back((*it));
            it ++;
        }
        return res;
    }
};
