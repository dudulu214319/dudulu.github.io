二叉树的层序遍历

2023/10/23 算法 树

二(n)叉树层序遍历

关键点

  • 采用节点指针对列q
  • q.size()作循环完成对一层节点的访问
  • 左右节点的放入队列的顺序可以颠倒，实现每一层的反转访问

代码

class Solution {
public:
    vector<vector<int>> levelOrder(Node* root) {
        vector<vector<int>> ans;
        if(!root) return ans;
        
        queue<Node*> q;
        q.push(root);

        while (!q.empty()) //整个算法终止条件
        {
            int cnt = q.size();
            vector<int> level;
            for (size_t i = 0; i < cnt; i++)
            {
                Node* cur = q.front();
                q.pop(); //弹出
                level.push_back(cur->val); //存值
                for (Node* child : cur->children) //有点类似pyhton里遍历迭代器的写法
                {
                    q.push(child);
                }                
            }
            ans.push_back(level);
        }
        return ans;
    }
};
