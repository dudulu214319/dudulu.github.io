二叉树DFS的应用

相同的树/对称的树

递归关键点

  • 入口
  • 出口
  • 如何借助分解一步将大问题分解成相同性质的子问题

注意，对一棵树来说叫对称，isf函数判断的是两棵树是否能对折

class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        if (!p && !q)
        {
            return true;
        }else if (!p || !q)
        {
            return false;
        }else if (p->val != q->val)
        {
            return false;
        }else{
            return isSameTree(p->left, q->left) && isSameTree(p->right, q->right);
        }  
    }
};

class Solution {
public:
    bool isf(TreeNode* p, TreeNode* q){
        if(!p && !q) return true;
        else if (!p || !q)
        {
            return false;
        }else if (p->val != q->val)
        {
            return false;
        }else {
            return isf(p->left, q->right) && isf(p->right, q->left);
        }
    }

    bool isSymmetric(TreeNode* root) {
        return isf(root->left, root->right);
    }
};

合并二叉树

关键点

  • 递归问题有时候可以先整体思考，一步设计为处理根节点，然后两边左子树合并左子树，右子树合并右子树，很自然的就是一个递归。
  • 处理两边根节点是否为空，直接俩if完事。。。
class Solution {
public:
    TreeNode* mergeTrees(TreeNode* root1, TreeNode* root2) {
        if(!root1) return root2;
        if(!root2) return root1;
        root1->val += root2->val;
        root1->left = mergeTrees(root1->left, root2->left);
        root1->right = mergeTrees(root1->right, root2->right);
        return root1;
    }
};

输出树的所有路径

  • 很容易用递归+DFS解决，先处理一下根节点，然后分别接上左树与右树的路径，都保存下来即可
class Solution {
public:
    vector<string> binaryTreePaths(TreeNode* root) {
        vector<string> ans;
        if(!root) return ans;
        string start = to_string(root->val);
        if(root->left) {
            for (string s : binaryTreePaths(root->left)) {
                ans.push_back(start + "->" + s);
            }
        }
        if(root->right) {
            for (string s : binaryTreePaths(root->right)) {
                ans.push_back(start + "->" + s);
            }
        }
        if(!root->left && !root->right) ans.push_back(start);
        return ans;
    }
};

判断树的路经总和有没有目标值

  • 递归，目标值不断减去根节点的值即可
  • 注意只有叶子节点才能返回true，半道上遇到val == targetSum的情况也只能是减到0，后面继续。
class Solution {
public:
    bool hasPathSum(TreeNode* root, int targetSum) {
        if(!root) return false;
        if(root->val == targetSum && !root->left && !root->right) return true;
        return hasPathSum(root->left, targetSum-root->val) || hasPathSum(root->right, targetSum-root->val);
    }
};

翻转二叉树

  • 直接递归解决即可，一步设计为根节点不动，左子树反转后连到根节点右边，右子树反转后连到根节点左边，以此分解成相同性质的子问题。

class Solution {
public:
    TreeNode* invertTree(TreeNode* root) {
        if(!root) return nullptr;
        TreeNode* r = invertTree(root->right);
        TreeNode* l = invertTree(root->left);
        root->left = r;
        root->right = l;
        return root;
    }
};

二叉树的深度问题

  • 比较简单，递归出口就是叶子节点的空子节点，一步就是加一层，然后左右子树取max就行。
class Solution {
public:
    int maxDepth(TreeNode* root) {
        if(!root) return 0;
        return std::max(maxDepth(root->left), maxDepth(root->right)) + 1;
    }
};

  • 这个题和之前有一点变化，递归终点不再是叶子节点的空子节点，而是叶子节点自己。所以一步要设计的大一点，需要判断左右子树的空情况。
  • 递归终点的设计与一步也是息息相关的。
class Solution {
public:
    int minDepth(TreeNode* root) {
        if(!root) return 0;
        if(!root->left && !root->right) return 1;
        if(!root->left) return minDepth(root->right) + 1;
        if(!root->right) return minDepth(root->left) + 1;
        return std::min(minDepth(root->left), minDepth(root->right)) + 1;
    }
};

  • 题倒不难，但是用max函数赋值的时间效率比作if判断要高，影响运行时长。。。以后取最大值还是优先用max函数。
class Solution {
public:
    int maxDepth(Node* root) {
        if(!root) return 0;
        int depth = 0;
        vector<Node*> c = root->children;
        for(auto node : c){
            // if(maxDepth(node) > depth) depth=maxDepth(node);
            depth = max(maxDepth(node), depth);
        }
        return depth + 1;
    }
};

判断平衡树

关键点

  • 如图所示，关键在于意识到对于树是否平衡的判断可以在计算树的高（深）度的递归过程中顺带实现。
  • 基于后序遍历

代码

class Solution {
public:
    int height(TreeNode* root) {
        if(!root) return 0;
        int left_h = height(root->left);
        int right_h = height(root->right);
        if(left_h == -1 || right_h == -1 || abs(left_h-right_h) > 1){ // 后面有-1就直接-1
            return -1;
        }
        return std::max(left_h, right_h) + 1;
    }

    bool isBalanced(TreeNode* root) {
        return height(root) > -1 ? true : false;
    }
};

二叉树的最近公共祖先

关键点

  • 分析问题，不难发现其实就是判断p和q在左右子树上的散落情况，可以用DFS去解决
  • 采用后序遍历的思路，先设计递归树逻辑，再设计代码。

代码

class Solution {
public:
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        if(!root) return nullptr;
        if(root == p) return root;
        if(root == q) return root;
        //后序遍历的形式
        TreeNode* left = lowestCommonAncestor(root->left, p, q); //避免反复调用递归
        TreeNode* right = lowestCommonAncestor(root->right, p, q);
        if(left && right) {
            return root;
        }
        if(left) return left;
        if(right) return right;
        return nullptr;
    }
};

⭐从前序与中序遍历序列构造二叉树

  • 可以理解为一道DFS的模板题目
  • 二叉树的DFS问题，大多是处理一步（处理根节点），然后递归处理左右子树，最后返回根节点。
  • 终止条件反而需要仔细推敲，一般是处理到返回NULL。
class Solution {
public:
    unordered_map<int, int> index;

    TreeNode* buildTree_Helpler (const vector<int>& preorder, const vector<int>& inorder, int preorder_left,
    int preorder_right, int inorder_left, int inorder_right) {
        if(preorder_left > preorder_right || inorder_left > inorder_right) {
            return nullptr;
        }

        // 前序遍历中的第一个节点就是根节点
        int preorder_root = preorder_left;
        // 在中序遍历中定位根节点位置
        int inorder_root = index[preorder[preorder_root]];

        // 先把根节点建立出来
        TreeNode* root = new TreeNode(preorder[preorder_root]);
        // 得到左子树中的节点数目
        int size_left_subtree = inorder_root - inorder_left;

        root->left = buildTree_Helpler(preorder, inorder, preorder_left + 1, preorder_left + size_left_subtree, inorder_left, inorder_root - 1);
        root->right = buildTree_Helpler(preorder, inorder, preorder_left + size_left_subtree + 1, preorder_right, inorder_root + 1, inorder_right);

        return root;
    }

    TreeNode* buildTree(vector<int>& preorder, vector<int>& inorder) {
        int n = preorder.size();
        for (int i = 0; i < n; ++i) {
            index[inorder[i]] = i;
        }
        return buildTree_Helpler(preorder, inorder, 0, n-1, 0, n-1);
    }
};
