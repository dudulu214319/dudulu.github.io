快速排序

int Partition(vector<int>& arr, int left, int right) {
    int pivot = arr[left];
    while (left < right) {
        while (left < right && arr[right] >= pivot) {
            right --;
        }
        arr[left] = arr[right];
        while (left < right && arr[left] <= pivot) {
            left ++;
        }
        arr[right] = arr[left];
    }
    arr[left] = pivot;
    return left;
}

void QuickSort(vector<int>& arr, int left, int right) {
    if(left < right){
        int pos = Partition(arr, left, right);
        QuickSort(arr, left, pos-1);
        QuickSort(arr, pos+1, right);
    }
}

vector<int> MySort(vector<int>& arr) {
    QuickSort(arr, 0, arr.size()-1);
    return arr;
}

归并排序

vector():创建一个空vector
vector(int nSize):创建一个vector,元素个数为nSize
vector(int nSize,const t& t):创建一个vector，元素个数为nSize,且值均为t

void Merge(vector<int>& arr, int start, int end, vector<int>& temp) {
    int left = start;
    int mid = (start+end) / 2;
    int right = mid + 1;
    int index = start;
    while (left <= mid && right <= end) {
        if(arr[left] < arr[right]) {
            temp[index ++] = arr[left ++]; 
        }else {
            temp[index ++] = arr[right ++];
        }
    }
    while (left <= mid) {
        temp[index ++] = arr[left ++];
    }
    while (right <= end) {
        temp[index ++] = arr[right ++];
    }
    for (int i = start; i <= end; i ++) {
        arr[i] = temp[i];
    }
}

void MergeSort(vector<int>& arr, int start, int end, vector<int>& temp) {
    if(start < end) {
        int mid = (start + end) / 2;
        MergeSort(arr, start, mid, temp);
        MergeSort(arr, mid + 1, end, temp);
        Merge(arr, start, end, temp);
    }
}

vector<int> MySort(vector<int>& arr) {
    vector<int> temp(arr.size());
    MergeSort(arr, 0, arr.size()-1, temp);
    return arr;
}

冒泡排序

  • 原理：从未排序部分的数列中挑出（冒泡过程不断更新最大值）最大的，放到已排序数列的最左边。
  • 注意两个for循环都有-1。
  • i循环设置flag可以提前退出。
vector<int> MySort(vector<int>& arr) {
        int n = arr.size();
        for(int i = 0;i < n - 1;i ++){
            int flag = 0;
            for(int j = 0;j < n - i - 1;j ++) {
                if(arr[j] > arr[j+1]) {
                    // int temp = arr[j];
                    // arr[j] = arr[j + 1];
                    // arr[j+1] = temp;
                    swap(arr[j], arr[j+1]);
                    flag = 1;
                }
            }
            if(!flag) return arr;
        }
        return arr;
    }

选择排序

  • 可以在两端分别设置begin++和end–逼近，同时更新Min和Max，但是注意begin==Max的特殊情况
  • while循环套for循环，两层循环。
vector<int> MySort(vector<int>& arr) {
    int n = arr.size();
    int begin = 0;
    int end = n - 1;
    int Min, Max;
    while (begin < end) {
        Min = begin;
        Max = begin;
        for(int i = begin;i <= end;i ++) {
            if(arr[i] < arr[Min]) Min = i;
            if(arr[i] > arr[Max]) Max = i;
        }
        swap(arr[begin], arr[Min]);
        if(begin == Max) Max = Min; //特殊情况
        swap(arr[end], arr[Max]);

        begin ++;
        end --;
    }
    return arr;
}

插入排序

  • i从1开始，就当后面的数不知道（抽牌），一个个往前面插入。（把小数插到前面去）
  • 插入过程需要先备份，再边往前检查边移动，检查到[j]不大于当前[i]的值时/j<0时，[j+1]=temp即可。
  • 如果arr[i-1]<arr[i]，就没必要进第二个循环。
  • PS：虽然简单但写起来很别扭。。。
vector<int> MySort(vector<int>& arr) {
        int n = arr.size();
        int temp;

        for(int i = 1;i < n;i ++) {
            if(arr[i-1] > arr[i]) { //判断是不是小数
                temp = arr[i];
                int j;
                for(j = i-1;j >= 0 && arr[j] > temp;j --) {
                    arr[j+1] = arr[j];
                }
                // arr[j] <= temp, arr[j+1] > temp
                // j = -1
                arr[j+1] = temp;
            }
        }
        return arr;
    }

  • if(arr[i-1] > arr[i])可以不做，更清晰。

希尔排序（插入排序的改进）

  • 大体上来说，gap有多大就有几个组。每个组里有几个数得看n/gap的值。

  • 最外层的循环从大gap开始逐渐变小，最后=1也就是普通的插入。

  • i在前移的过程中不断切换所处的组。

  • 数组大小奇数偶数？

    vector<int> MySort(vector<int>& arr) {
    int n = arr.size();
    int gap = n / 2;
    while (gap >= 1) {
        for(int i = 0;i < n - gap;i ++) {
            int temp = arr[i];
            int j = i - gap;
            while (j >= 0 && arr[j] > arr[i]) {
                arr[j + gap] = arr[j];
                j -= gap;
            }
            arr[j + gap] = temp;
        }
        gap /= 2;
    }
    return arr;
}

    vector<int> MySort(vector<int>& arr) {
    int n = arr.size();
    int gap = n;
    while (gap > 1)
    {
        gap /= 2;
        // gap = gap / 3 + 1;
        for (int i = 0; i < n - gap; i++)
        {
            int end = i;
            int x = arr[end + gap];
            while (end >= 0)
            {
                if (arr[end] > x)
                {
                    arr[end + gap] = arr[end];
                    end -= gap;
                }
                else
                {
                    break;
                }
            }
            arr[end + gap] = x;
        }
    }
        return arr;
    }
};

堆排序

堆的基本性质

  • 堆通常用作实现优先队列，大顶堆相当于元素按从大到小顺序出队的优先队列。从使用角度来看，我们可以将“优先队列”和“堆”看作等价的数据结构。
/* 初始化堆 */
// 初始化小顶堆
priority_queue<int, vector<int>, greater<int>> minHeap;
// 初始化大顶堆
priority_queue<int, vector<int>, less<int>> maxHeap;

/* 元素入堆 */
maxHeap.push(1);
maxHeap.push(3);
maxHeap.push(2);
maxHeap.push(5);
maxHeap.push(4);

/* 获取堆顶元素 */
int peek = maxHeap.top(); // 5

/* 堆顶元素出堆 */
// 出堆元素会形成一个从大到小的序列
maxHeap.pop(); // 5
maxHeap.pop(); // 4
maxHeap.pop(); // 3
maxHeap.pop(); // 2
maxHeap.pop(); // 1

/* 获取堆大小 */
int size = maxHeap.size();

/* 判断堆是否为空 */
bool isEmpty = maxHeap.empty();

/* 输入列表并建堆 */
vector<int> input{1, 3, 2, 5, 4};
priority_queue<int, vector<int>, greater<int>> minHeap(input.begin(), input.end());

回到堆排序

  • 原理：一个arr数组可以平铺到一棵完全二叉树里，一一对应的关系，所以我们可以把arr数组视为一个堆，并使用堆的性质排序。
  • 本质是一种选择排序，不断地从未排序序列里选出最大值放在arr[0]，然后和已排序序列首部的前一个数交换。这样大顶堆可以实现从小到大排序，小顶堆可以实现从大到小排序。
  • 大/小顶堆本身不是一种很强的数据结构，只能保证上面的数比下面的大/小，左右没有限制。
  • 下面的gif图有问题，没有完成大顶堆化。

class Solution {
    public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 将给定数组排序
     * @param arr int整型vector 待排序的数组
     * @return int整型vector
     */
    void siftDown(vector<int>& arr, int n, int i) {
        while (true) {
            int l = 2*i + 1;
            int r = 2*i + 2;
            int ma = i;
            if(l <= n-1 && arr[l] > arr[ma]) ma = l;
            if(r <= n-1 && arr[r] > arr[ma]) ma = r;
            if(ma == i) break;
            swap(arr[i], arr[ma]);
            // 堆化索引继续向下,以此维持未排序部分是大顶堆
            i = ma;
        }
    }

    vector<int> MySort(vector<int>& arr) {
        int n = arr.size();
        // 大顶堆化整个arr，从倒数第一个非叶子节点开始
        for (int i = n/2 - 1; i >= 0; i --) {
            siftDown(arr, n, i);
        }
        for(int i = n - 1;i >= 0;i --) {
            swap(arr[0], arr[i]);
            // 不断维持未排序部分是大顶堆，i变小限制堆的视野
            siftDown(arr, i, 0);
        }
        return arr;
    }
};
