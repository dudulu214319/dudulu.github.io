⭐206-反转链表

class Solution {
public:
    // 1.三指针折一次法
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) { return nullptr; }
        ListNode* pre = nullptr;
        ListNode* cur = head;
        while (cur != nullptr)
        {
            ListNode* next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
    // 2.双指针+头插法
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) { return nullptr; }
        ListNode ans(-1);
        ListNode* cur = head;
        while (cur != nullptr)
        {
            ListNode* next = cur->next; // 备份
            cur->next = ans.next; // 先接尾
            ans.next = cur; // cur插到ans后面
            cur = next;
        }
        
        return ans.next;
    }
};

92-指定区间反转

  • p|cn -> pn|c模型:

    cur->next = next->next;

​ next->next = pre->next; || pre->next = next; // 把n接到p后面时，先接尾更好，不牵扯cur

​ pre->next = next; || next.next = cur;

class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        if(!head)return nullptr;
        ListNode dummy(-1);
        dummy.next = head;
        ListNode* pre = &dummy;
        for (int i = 0; i < left - 1; i++)
        {
            pre = pre->next; // 到达区间左端的左一个
        }
        ListNode* cur = pre->next;
        for (int j = left; j < right; j++) // cur把next拱到pre后面
        {
            ListNode* next = cur->next;
            cur->next = next->next;
            next->next = pre->next;
            pre->next = next;
        }
        return dummy.next;
    }
};

24-两两交换链表中的节点

class Solution {
    public ListNode swapPairs(ListNode head) {
        // p|cn -> pn|c交换模型
        ListNode dummyNode = new ListNode(-1);
        dummyNode.next = head;
        ListNode pre = dummyNode;
        while (pre.next != null && pre.next.next != null) {
            ListNode cur = pre.next;
            ListNode next = pre.next.next;
            cur.next = next.next;
            next.next = pre.next; // 统一先连尾
            pre.next = next; // 再连头
            pre = cur; // pre这个引用移动成为下一个一对的前置
        }
        return dummyNode.next;
    }
}

234-回文链表

  • 快慢指针+一半反转
  • 需要分奇数偶数情况讨论，slow会停在中间+1（偶数）或者正中（奇数）；fast停在null（偶数）或尾部（奇数）。
  • 真考试不如直接用栈（不要求空间的话）
class Solution {
    public boolean isPalindrome(ListNode head) {
        if(head == null || head.next == null) {
            return true;
        }
        ListNode slow = head;
        ListNode fast = head;
        ListNode cur = head;
        ListNode pre = null;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            cur.next = pre; // 三指针折一次
            pre = cur;
            cur = slow;
        }
        cur = pre; // cur多退一步
        if(fast != null) { // 奇数情况，slow在正中需要再走一步
            slow = slow.next;
        }
        while (cur != null && slow != null) {
            if(cur.val != slow.val) {
                return false;
            }
            cur = cur.next;
            slow = slow.next;
        }
        return true;
    }
}
