⭐206-反转链表

class Solution {
public:
    // 1.三指针折一次法
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) { return nullptr; }
        ListNode* pre = nullptr;
        ListNode* cur = head;
        while (cur != nullptr)
        {
            ListNode* next = cur->next;
            cur->next = pre;
            pre = cur;
            cur = next;
        }
        return pre;
    }
    // 2.双指针+头插法
    ListNode* reverseList(ListNode* head) {
        if (head == nullptr) { return nullptr; }
        ListNode ans(-1);
        ListNode* cur = head;
        while (cur != nullptr)
        {
            ListNode* next = cur->next; // 备份
            cur->next = ans.next; // 先接尾
            ans.next = cur; // cur插到ans后面
            cur = next;
        }
        
        return ans.next;
    }
};

92-指定区间反转

  • p|cn -> pn|c模型:

    cur->next = next->next;

​ next->next = pre->next; || pre->next = next; // 把n接到p后面时，先接尾更好，不牵扯cur

​ pre->next = next; || next.next = cur;

class Solution {
public:
    ListNode* reverseBetween(ListNode* head, int left, int right) {
        if(!head)return nullptr;
        ListNode dummy(-1);
        dummy.next = head;
        ListNode* pre = &dummy;
        for (int i = 0; i < left - 1; i++)
        {
            pre = pre->next; // 到达区间左端的左一个
        }
        ListNode* cur = pre->next;
        for (int j = left; j < right; j++) // cur把next拱到pre后面
        {
            ListNode* next = cur->next;
            cur->next = next->next;
            next->next = pre->next;
            pre->next = next;
        }
        return dummy.next;
    }
};

24-两两交换链表中的节点

class Solution {
    public ListNode swapPairs(ListNode head) {
        // p|cn -> pn|c交换模型
        ListNode dummyNode = new ListNode(-1);
        dummyNode.next = head;
        ListNode pre = dummyNode;
        while (pre.next != null && pre.next.next != null) {
            ListNode cur = pre.next;
            ListNode next = pre.next.next;
            cur.next = next.next;
            next.next = pre.next; // 统一先连尾
            pre.next = next; // 再连头
            pre = cur; // pre这个引用移动成为下一个一对的前置
        }
        return dummyNode.next;
    }
}

234-回文链表

  • 快慢指针+一半反转
  • 需要分奇数偶数情况讨论，slow会停在中间+1（偶数）或者正中（奇数）；fast停在null（偶数）或尾部（奇数）。
  • 真考试不如直接用栈（不要求空间的话）
class Solution {
    public boolean isPalindrome(ListNode head) {
        if(head == null || head.next == null) {
            return true;
        }
        ListNode slow = head;
        ListNode fast = head;
        ListNode cur = head;
        ListNode pre = null;
        while (fast != null && fast.next != null) {
            slow = slow.next;
            fast = fast.next.next;
            cur.next = pre; // 三指针折一次
            pre = cur;
            cur = slow;
        }
        cur = pre; // cur多退一步
        if(fast != null) { // 奇数情况，slow在正中需要再走一步
            slow = slow.next;
        }
        while (cur != null && slow != null) {
            if(cur.val != slow.val) {
                return false;
            }
            cur = cur.next;
            slow = slow.next;
        }
        return true;
    }
}

141-环形链表

  • 思路一：unordered_set 不需要记录键值对，只需要记录值
  • 思路二：龟兔赛跑，快慢指针
class Solution {
public:
    bool hasCycle(ListNode *head) {
        if(head == nullptr)
        return false;

        //双指针追击的思路，龟兔赛跑思想
        ListNode * i = head;
        ListNode * j = head;

        while (1){
            if(i->next == nullptr)return false;
            if(i->next->next == nullptr)return false;
            i = i->next->next;
            if(i == j)return true;
            j = j->next;
        }
    }
};

class Solution {
public:
    bool hasCycle(ListNode *head) {
        unordered_set<ListNode*> seen;
        while (head != nullptr) {
            if (seen.count(head)) {
                return true;
            }
            seen.insert(head);
            head = head->next;
        }
        return false;
    }
};

142-环形链表Ⅱ

  • 双指针两次相遇法
    • 第一次相遇：x+b,x
    • fast置于head（特殊情况，第一次相遇位置是head，说明整条链表是个环）
    • 第二次相遇：x+b+a，x+a在入口相遇
class Solution {
public:
    ListNode *detectCycle(ListNode *head) {
        if (head == nullptr) return nullptr;
        if (head->next == nullptr) return nullptr;
        ListNode* f = head;
        ListNode* s = head;
        while (true)
        {
            f = f->next->next;
            s = s->next;
            if(f == s)break; //有环
            if(!f || !f->next) return nullptr;
        }
        if (f == head) // 特殊情况，整个链表是个环
        {
            return head;
        }
        f = head;
        while (true)
        {
            f = f->next;
            s = s->next;
            if(f == s) return f; // 第二次相撞，x+b+a,x+a必然在入口停止
        }
    }
};

class Node { // 双向链表节点
public:
    int key, value;
    Node *prev, *next;
    Node(int k = 0, int v = 0) : key(k), value(v) {}
};
  • 私有变量
    • unordered_map<int, Node*> key_to_node; // key-Node*哈希表
    • Node* dummy; //头节点
    • int capacity; // 容量
  • 私有方法
    • void remove(Node* x)
    • void push_front(Node* x)
    • Node* get_node(int key)
  • 公共方法
    • 构造函数 dummy构成环形
    • int get()直接调用get_node
    • void put(key, value)
      • 检查有无，是否要覆盖auto
      • New节点
      • 维护哈希
      • push_front
      • 如果超过容量
        • 取dummy->prev
        • 删哈希
        • 删节点
        • delete释放内存
class LRUCache {
private:
    unordered_map<int, Node*> key_to_node; // key-Node*哈希表
    Node* dummy; //头节点
    int capacity; // 容量

    // 删除一个节点（抽出一本书）
    void remove(Node* x) {
        // Node* x = key_to_node[key];
        x->prev->next = x->next;
        x->next->prev = x->prev;
    }

    // 在链表头添加一个节点（把一本书放在最上面）
    void push_front(Node* x) {
        x->prev = dummy;
        x->next = dummy->next;
        dummy->next = x;
        x->next->prev = x;
    }

    // 根据key查找【节点指针】，不存在返回null，以及顺带着remove和push
    Node* get_node(int key) {
        auto it = key_to_node.find(key); // count?
        if(it == key_to_node.end()) {
            return nullptr;
        }
        auto node = it->second;
        remove(node);
        push_front(node);
        return node;
    }

public:
    LRUCache(int capacity): capacity(capacity), dummy(new Node()) {
        // this->capacity = capacity;
        // dummy = new Node();
        dummy->prev = dummy;
        dummy->next = dummy;
    }
    
    int get(int key) {
        if(get_node(key)) {
            return get_node(key)->value;
        }else {
            return -1;
        }
    }
    
    void put(int key, int value) { // 逐步维护链表和哈希表
        auto node = get_node(key); // auto承接可能的空
        if (node)
        {
            node->value = value;
            return;
        }
        Node* new_node = new Node(key, value); // new指针
        key_to_node[key] = new_node; // 维护哈希
        push_front(new_node);
        if (key_to_node.size() > capacity)
        {
            auto back_node = dummy->prev;
            key_to_node.erase(back_node->key); // 维护哈希
            remove(back_node);
            delete back_node; // 没有再放到顶的操作
        }
    }
};
