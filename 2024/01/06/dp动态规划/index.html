动态规划

核心思路：

  • 确定数组和f[i]状态定义
  • 从尾部分析一步，怎么拆解子问题，写出状态转移方程
  • 从头部（初始条件）开始循环计算至尾部结果（或者计算完整个数组，从里面再挑出结果）

62-不同路径

  • 关键在于理解dp的无后效性/类马尔科夫性，一个状态值的计算只与某些状态的当前值有关，而不关心前面的值是咋得来的。
  • 这类题目（有的题目不行）还有空间上的近邻性可以利用，从二维数组压缩到一维滚动数组。
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<int> dp(n);
        for (int i = 0; i < n; i++)
        {
            dp[i] = 1;
        }
        for (int i = 0; i < m - 1; i++)
        {
            for (int j = 1;j < n;j ++) {
                dp[j] += dp[j - 1]; 
            }
        }
        return dp[n - 1];
    }
};

64-最小路径和

  • 思路和上题大差不差，我还是用一维滚动数组做的
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> g(n); // 一维滚动数组
        int Min;
        g[0] = grid[0][0];
        for (int i = 1; i < n; i++)
        {
            g[i] = g[i - 1] + grid[0][i];
        }
        for (int i = 1; i < m; i++) // 从第二行开始
        {
            // 第一列：grid对应值+g当前值
            g[0] = g[0] + grid[i][0];
            for (int j = 1;j < n;j ++) {
                Min = g[j - 1] < g[j] ? g[j - 1] : g[j]; // 确定小值
                g[j] = Min + grid[i][j];
            }
        }
        return g[n - 1]; // 递归树的起点
    }
};

120-三角形最小路径和

  • 思路还是差不多，我这里用了二维滚动数组。（近邻性）
class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        int n = triangle.size();
        if(n == 1) return triangle[0][0];
        vector<vector<int>> f(2, vector<int>(n, 0)); // 只用两层vector就够了，做滚动即可
        int Min;
        int res;
        f[0][0] = triangle[0][0];
        for (int i = 1; i < n; i ++)
        {
            for (int j = 0; j < i + 1; j ++)
            {
                if(j == 0) {
                    f[1][j] = triangle[i][j] + f[0][0];
                }else if(j == i) {
                    f[1][j] = triangle[i][j] + f[0][j-1];
                }else {
                    Min = f[0][j] < f[0][j-1] ? f[0][j] : f[0][j-1];
                    f[1][j] = triangle[i][j] + Min;
                }
            }
            if(i == n - 1) {
                res = *(min_element(f[1].begin(), f[1].end()));
                return res;
            }
            for (int k = 0; k < i + 1; k++)
            {
                f[0][k] = f[1][k];
            }
        }
        return res;
    }
};
