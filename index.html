动态规划+回溯

核心思路：

  • 确定数组和f[i]状态定义
  • 从尾部分析一步，怎么拆解子问题，写出状态转移方程
  • 从头部（初始条件）开始循环计算至尾部结果（或者计算完整个数组，从里面再挑出结果）

62-不同路径

  • 关键在于理解dp的无后效性/类马尔科夫性，一个状态值的计算只与某些状态的当前值有关，而不关心前面的值是咋得来的。
  • 这类题目（有的题目不行）还有空间上的近邻性可以利用，从二维数组压缩到一维滚动数组。
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<int> dp(n);
        for (int i = 0; i < n; i++)
        {
            dp[i] = 1;
        }
        for (int i = 0; i < m - 1; i++)
        {
            for (int j = 1;j < n;j ++) {
                dp[j] += dp[j - 1]; 
            }
        }
        return dp[n - 1];
    }
};

64-最小路径和

  • 思路和上题大差不差，我还是用一维滚动数组做的
class Solution {
public:
    int minPathSum(vector<vector<int>>& grid) {
        int m = grid.size();
        int n = grid[0].size();
        vector<int> g(n); // 一维滚动数组
        int Min;
        g[0] = grid[0][0];
        for (int i = 1; i < n; i++)
        {
            g[i] = g[i - 1] + grid[0][i];
        }
        for (int i = 1; i < m; i++) // 从第二行开始
        {
            // 第一列：grid对应值+g当前值
            g[0] = g[0] + grid[i][0];
            for (int j = 1;j < n;j ++) {
                Min = g[j - 1] < g[j] ? g[j - 1] : g[j]; // 确定小值
                g[j] = Min + grid[i][j];
            }
        }
        return g[n - 1]; // 递归树的起点
    }
};

120-三角形最小路径和

  • 思路还是差不多，我这里用了二维滚动数组。（近邻性）
class Solution {
public:
    int minimumTotal(vector<vector<int>>& triangle) {
        int n = triangle.size();
        if(n == 1) return triangle[0][0];
        vector<vector<int>> f(2, vector<int>(n, 0)); // 只用两层vector就够了，做滚动即可
        int Min;
        int res;
        f[0][0] = triangle[0][0];
        for (int i = 1; i < n; i ++)
        {
            for (int j = 0; j < i + 1; j ++)
            {
                if(j == 0) {
                    f[1][j] = triangle[i][j] + f[0][0];
                }else if(j == i) {
                    f[1][j] = triangle[i][j] + f[0][j-1];
                }else {
                    Min = f[0][j] < f[0][j-1] ? f[0][j] : f[0][j-1];
                    f[1][j] = triangle[i][j] + Min;
                }
            }
            if(i == n - 1) {
                res = *(min_element(f[1].begin(), f[1].end()));
                return res;
            }
            for (int k = 0; k < i + 1; k++)
            {
                f[0][k] = f[1][k];
            }
        }
        return res;
    }
};

32-最长有效括号

int longestValidParentheses(string s) {
        int maxans = 0, n = s.length();
        vector<int> dp(n, 0);
        for (int i = 1; i < n; i++) {
            if (s[i] == ')') {
                if (s[i - 1] == '(') {
                    dp[i] = (i >= 2 ? dp[i - 2] : 0) + 2;
                } else if (i - dp[i - 1] > 0 && s[i - dp[i - 1] - 1] == '(') { // 得保证1号三角位置存在
                    dp[i] = dp[i - 1] + ((i - dp[i-1]) >= 2 ? dp[i - dp[i-1] - 2] : 0) + 2; // 一号位置存在的基础上加上可能的2号三角位置带出来的串
                }
                // 1号位置如果是')'则必然对应f值为0，否则会被包含在f[i-1]里
                maxans = max(maxans, dp[i]);
            }
        }
        return maxans;
}

22.括号生成

class Solution {
public:
    void dfs(int left, int right, string path, int n, vector<string>& res) {
        if(path.size() == n*2) {
            res.push_back(path);
            return;
        }
        if(left < n) { // 把待定变为确定的时候可以判断剪枝
            path.push_back('(');
            dfs(left + 1, right, path, n, res);
            path.pop_back();
        }
        if(right < left) { // 剪枝
            path.push_back(')');
            dfs(left, right + 1, path, n, res);
            path.pop_back();
        }
    }

    vector<string> generateParenthesis(int n) {
        vector<string> res;
        string path;
        dfs(0, 0, path, n, res);
        return res;
    }
};

77.组合

class Solution {
public:
    // 范围参数，终止条件参数，状态参数*(传形参，很自然外层的dfs不会更新path)，结果参数
    void dfs(int n, int start, int k, vector<int> path, vector<vector<int>> &res) {
        if(path.size() == k){ // 长度为K终止
            res.push_back(path);
            return;
        }
        for (int i = start; i <= n; i++) // 一横，遍历当下可选的【暂定】，同时做了范围缩减
        {
            path.push_back(i); // 更新状态，对于下一级来说这就是【确定】
            dfs(n, i + 1, k, path, res); // 一纵，+1缩减范围
            path.pop_back();
        }
    }

    vector<vector<int>> combine(int n, int k) {
        vector<vector<int>> res;
        vector<int> path;
        dfs(n, 1, k, path, res);
        return res;
    }
};

131.分割回文串

  • 本质依然是组合问题，组合的是【切割】
  • 终止条件是当下【切割位置是最后一个切割】
  • 注意剪枝
class Solution {
public:
    bool isPalindrome(string s, int start, int end) {
        for(int i = start, j = end;i < j;i ++, j --) {
            if(s[i] != s[j]) {
                return false;
            }
        }
        return true;
    }

    void dfs(string s, int start, vector<string> path, vector<vector<string>>& res) {
        if(start >= s.length()) { // 终止条件是当下切割位置是最后一个切割
            res.push_back(path);
            return;
        }
        for (int i = start; i < s.length(); i++) // 范围选定一横
        {
            if(isPalindrome(s, start, i)) { // 准备把暂定变为确定放入path时先做判断，剪枝
                string m = s.substr(start, i - start + 1);
                path.push_back(m);
                dfs(s, i + 1, path, res); // 一纵
                path.pop_back();
            }
        }
    }

    vector<vector<string>> partition(string s) {
        vector<vector<string>> res;
        vector<string> path;
        dfs(s, 0, path, res); //
        return res;
    }
};

⭐132.最少分割回文串

  • 用中心生成的思路完成isPalin上三角矩阵。
  • f[i]设定为把以s[i]为结尾的串最少分割成全是回文子串的子串数量。
  • min/max是确定最后一个状态由谁转移而来，每一个f[i]值的确定都是一个循环+迭代更新出值的过程。
class Solution {
public:
    int minCut(string s) {
        int n = s.size();
        vector<vector<int>> isPalin(n, vector<int>(n));
        vector<int> f(n);
        for (int i = 0; i < n; i++)
        {
            for (int j = i; j < n; j++)
            {
                isPalin[i][j] = false;
            }
        }
        int l, r;
        for (int t = 0; t < n; t++)
        {
            // 奇数长度
            l = t;
            r = t;
            while (l>=0 && r < n && s[l] == s[r])
            {
                isPalin[l][r] = true;
                l --;
                r ++;
            }
            // 偶数长度
            l = t;
            r = t + 1;
            while (l>=0 && r < n && s[l] == s[r])
            {
                isPalin[l][r] = true;
                l --;
                r ++;
            }
        }
        f[0] = 1; // 1.f[i]设定为把以s[i]为结尾的串最少分割成全是回文子串的子串数量
        for (int i = 1; i < n; i++)
        {
            f[i] = INT16_MAX;
            for (int j = 0; j <= i; j++)
            {
                if(isPalin[j][i]) {
                    if(j > 0) f[i] = min(f[i], f[j-1]+1); // 2.不能确定最后一个f[i]状态由谁转移而来，所以for循环穷举所有可能的情况，迭代更新出最小值
                    else f[i] = 1;
                }
            }
        }
        return f[n-1]-1; // 切割数少一个
    }
};

647.回文子串

  • 中心扩展，找规律即可
class Solution {
public:
    int countSubstrings(string s) {
        int n = s.size(), ans = 0;
        for (int i = 0; i < 2 * n - 1; ++i) { // 以2n-1为循环，每次确定l和r
            int l = i / 2, r = i / 2 + i % 2;
            while (l >= 0 && r < n && s[l] == s[r]) {
                --l;
                ++r;
                ++ans;
            }
        }
        return ans;
    }
};

1143.最长公共子序列

class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        int m = text1.size();
        int n = text2.size();
        vector<vector<int>> f(m+1, vector<int>(n+1));
        for (int i = 0; i <= m; i++)
        {
            f[i][0] = 0;
        }
        for (int i = 0; i <= n; i++)
        {
            f[0][i] = 0;
        }
        for (int i = 1; i <= m; i++)
        {
            for (int j = 1; j <= n; j++)
            {
                f[i][j] = max(f[i-1][j], f[i][j-1]);
                if(text1[i-1] == text2[j-1]) { // 加了一层padding
                    f[i][j] = (f[i][j] < f[i-1][j-1]+1) ? f[i-1][j-1]+1 : f[i][j];
                }
            }
        }
        return f[m][n];
    }
};

72.编辑距离

class Solution {
public:
    int minDistance(string word1, string word2) {
        int m = word1.size();
        int n = word2.size();
        if(m == 0) return n;
        if(n == 0) return m;
        vector<vector<int>> f(m, vector<int>(n));
        for (int i = 0; i < m; i++)
        {
            if(word1[i] == word2[0]) {
                while (i < m) // 有相等字符之后就是i，之前就是i+1
                {
                    f[i][0] = i;
                    i ++;
                }
                break;
            }else{
                f[i][0] = i + 1;
            }
        }
        for (int i = 0; i < n; i++)
        {
            if(word1[0] == word2[i]) {
                while (i < n)
                {
                    f[0][i] = i;
                    i ++;
                }
                break;
            }else{
                f[0][i] = i + 1;
            }
        }
        for (int i = 1; i < m; i++)
        {
            for (int j = 1; j < n; j++)
            {
                f[i][j] = min(f[i-1][j]+1, f[i][j-1]+1); // 增删
                if(word1[i] == word2[j]) {
                    f[i][j] = min(f[i][j], f[i-1][j-1]); //改
                }else{
                    f[i][j] = min(f[i][j], f[i-1][j-1]+1); // 改
                }
            }
        }
        return f[m-1][n-1];
    }
};

152.乘积最大子数组

  • 主要是数字的正负不确定，所以可能要×前面的最大值/最小值才能得到当下的最大值/最小值。
class Solution {
public:
    int maxProduct(vector<int>& nums) {
        int n = nums.size();
        vector<int> f(n);
        vector<int> g(n);
        f[0] = nums[0];
        g[0] = nums[0];
        for (int i = 1; i < n; i++)
        {
            f[i] = max(max(nums[i], f[i-1]*nums[i]), g[i-1]*nums[i]);  // i为结尾设计状态
            g[i] = min(min(nums[i], g[i-1]*nums[i]), f[i-1]*nums[i]);  // i为结尾设计状态
        }
        return *max_element(f.begin(), f.end());
    }
};

53-最大子数组和

class Solution {
public:
    int maxSubArray(vector<int>& nums) {
        int n = nums.size();
        vector<int> ans(n);
        ans[0] = nums[0];
        int res = ans[0];
        for (int i = 1; i < n; i++)
        {
            ans[i] = max(nums[i], ans[i-1] + nums[i]); // i为结尾设计状态
            if(res < ans[i]) {
                res = ans[i];
            }
        }
        return res;
    }
};
